void main() {
	var a = stdin.readLineSync()!.replaceAll("szczupak ", "").split(":");
	var c = 0;
	
	c += 100*20*12*int.parse(a[0]);
	c += 20*12*int.parse(a[1]);
	c += 12*int.parse(a[2]);
	c += int.parse(a[3]);

}

Rozwiązanie w języku Rust:

use std::io;

fn main() {

	let mut a = String::new();

	io::stdin().read_line(&mut a);
	a = a.replace("szczupak ", "");

	let v: Vec<&str> = a.split(":").collect();
	let m: i32 = v[0].trim().parse().expect("m");
	let z: i32 = v[1].trim().parse().expect("z");
	let s: i32 = v[2].trim().parse().expect("s");
	let d: i32 = v[3].trim().parse().expect("d");
	let mut cena: i32 = 0;

	cena=cena+100*20*12*m;
	cena=cena+20*12*z;
	cena=cena+12*s;
	cena=cena+d;

	println!("{}", cena);
}
