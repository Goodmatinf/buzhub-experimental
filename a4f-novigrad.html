Rozwiązanie w języku Dart:

import 'dart:io';

void main() {
	var a = stdin.readLineSync()!.split(":");
	var m = int.parse(a[0]);
	var z = int.parse(a[1]);
	var s = int.parse(a[2]);
	var d = int.parse(a[3]);
	var y = 0;
	var c = 0;
	
	if (d > 11) {
		y = d%12;
		c = (d-y) ~/ 12;
		s = s + c;
		d = d - 12*c;
	}
	if (s > 19) {
		y = s % 20;
		c = (s - y) ~/ 20;
		z = z + c;
		s = s - 20*c;
	}
	if (z > 99) {
		y = z % 100;
		c = (z-y) ~/ 100;
		m = m + c;
		z = z - 100*c;
	}
	print("${m}:${z}:${s}:${d}");
}

Rozwiązanie w języku Rust:

use std::io;

fn main() {

	let mut a = String::new();

	io::stdin().read_line(&mut a);

	let v: Vec<&str> = a.split(":").collect();
	let mut m: i32 = v[0].trim().parse().expect("m");
	let mut z: i32 = v[1].trim().parse().expect("z");
	let mut s: i32 = v[2].trim().parse().expect("s");
	let mut d: i32 = v[3].trim().parse().expect("d");

	if d > 11 {
		let mut y = d%12;
		let mut c = (d-y) / 12;
		s += c;
		d -= 12*c;
	}

	if s > 19 {
		let mut y = s%20;
		let mut c = (s-y) / 20;
		z += c;
		s -= 20*c;
	}

	if z > 99 {
		let mut y = z%100;
		let mut c = (z-y) / 100;
		m += c;
		z -= 100*c;
	}

	println!("{}:{}:{}:{}", m, z, s, d);
}
