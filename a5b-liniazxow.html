a5b - Linia z x-ow

Status zadania: Pełna poprawność

Uwagi: Możliwe krótsze rozwiązanie



Przykładowe rozwiązanie:

a = input()
a = int(a)
przed = ""
for i in range(a):
	przed = przed + "xy"
print(przed)


Rozwiązanie w języku Dart:

import 'dart:io';

void main() {
	var a = int.parse(stdin.readLineSync()!);
	var c = "";
	
	for (var i = 1; i <= a; i++) {
		c += "xy";
	}
	
	print(c);
}

Rozwiązanie w języku Rust:

use std::io;

fn main() {

let mut a = String::new();

io::stdin().read_line(&mut a);

let a: i32 = a.trim().parse().expect("a");

let mut s =  String::new();

for i in 0..a {
s = s.to_owned() + "xy";
}

println!("{}", s);
}
