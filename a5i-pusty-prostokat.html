Rozwiązanie w języku Dart:

import 'dart:io';

void main() {
	var a = int.parse(stdin.readLineSync()!);
	
	for (var i = 0; i < a; i++) {
		if (i+1==1 || i+1==a) {
			print("x"*(2*a));
		}
		else {
			print("x${"."*(2*a-2)}x");
		}
	}
}
